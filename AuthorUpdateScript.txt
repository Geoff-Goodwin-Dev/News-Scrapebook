#!/bin/sh

git filter-branch --env-filter '

OLD_EMAIL="<geoffsmacbook@Geoffss-MacBook-Pro.local>"
CORRECT_NAME="Geoff-Goodwin-Dev"
CORRECT_EMAIL="<ggoodwin.dev@gmail.com>"

if [ "$GIT_COMMITTER_EMAIL" = "$<geoffsmacbook@Geoffss-MacBook-Pro.local>" ]
then
    export GIT_COMMITTER_NAME="$Geoff-Goodwin-Dev"
    export GIT_COMMITTER_EMAIL="<$ggoodwin.dev@gmail.com>"
fi
if [ "$GIT_AUTHOR_EMAIL" = "$<geoffsmacbook@Geoffss-MacBook-Pro.local>" ]
then
    export GIT_AUTHOR_NAME="$Geoff-Goodwin-Dev"
    export GIT_AUTHOR_EMAIL="$<ggoodwin.dev@gmail.com>"
fi
' --tag-name-filter cat -- --branches --tags
